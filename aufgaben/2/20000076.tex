\begin{align*}
P(S'_1|S_1) &= 0       & P(S'_1|S_2) &= 0 & P(S'_1|S_3) &= \frac12 \\
P(S'_2|S_1) &= \frac12 & P(S'_2|S_2) &= 0 & P(S'_2|S_3) &= \frac12 \\

H

\begin{pmatrix}
0           & 0 & 1           \\
